 fun getGreetingFromDate(date: java.util.Date): String {
     val calendar = java.util.Calendar.getInstance()
     calendar.time = date
     val hour = calendar.get(java.util.Calendar.HOUR_OF_DAY)
     return when (hour) {
         in 0..11 -> "Good Morning!"
         in 12..17 -> "Good Afternoon!"
         else -> "Good Evening!"
     }
 }
